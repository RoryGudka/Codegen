```typescript
import * as fs from 'fs';
import * as path from 'path';

function getCurrentDirectoryFileStructure(dirPath: string = __dirname): object {
  const fileStructure = {};

  function readDirectory(currentPath: string): object {
    const structure = {};
    const items = fs.readdirSync(currentPath);

    items.forEach(item => {
      const itemPath = path.join(currentPath, item);
      const stats = fs.statSync(itemPath);

      if (stats.isDirectory()) {
        structure[item] = readDirectory(itemPath);
      } else {
        structure[item] = 'file';
      }
    });

    return structure;
  }

  fileStructure['directory'] = readDirectory(dirPath);
  return fileStructure;
}

console.log(getCurrentDirectoryFileStructure());
```

This function `getCurrentDirectoryFileStructure` reads the files and directories within the current directory (`__dirname`) and returns a structured representation of the entire directory structure. It uses a recursive approach to read each directory and constructs a nested object representing the directory structure.
